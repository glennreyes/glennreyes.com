```js
import React from 'react';

const App = () => {
  const [date, setDate] = React.useState(new Date().toDateString());

  React.useEffect(() => {
    const t = setInterval(
      () => setDate(new Date().toDateString()),
      1000 * 60 * 24,
    );

    return () => clearInterval(t);
  });

  return <div>Date today: {date}</div>;
};
```
